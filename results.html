// /js/report.js import { col, getAll } from './database.js'; const $ = id => document.getElementById(id); let store = { classes: [], students: [], reports: [] }; async function loadAll() { [store.classes, store.students, store.reports] = await Promise.all([ getAll(col.classes), getAll(col.students), getAll(col.report_cards) ]); } function fillClassFilter() { const sel = $('classFilter'); sel.innerHTML = `` + store.classes.map(c => ``).join(''); } function renderTable(list) { const tbody = $('resultsTable').querySelector('tbody'); tbody.innerHTML = ''; if (!list.length) { tbody.innerHTML = `No reports found.`; return; } for (const r of list) { const stu = store.students.find(s => s.id === r.student_id) || {}; const cls = store.classes.find(c => c.id === r.class_id) || {}; const tr = document.createElement('tr'); tr.innerHTML = ` ${r.admission_no} ${stu.first_name||''} ${stu.last_name||''} ${cls.name||'-'} ${r.total_marks} ${r.mean_score} ${r.grade} ${(r.weak_subjects||[]).join(', ')} ${r.remark} `; tbody.appendChild(tr); } } window.printSingle = function(reportId) { const r = store.reports.find(x => x.id === reportId); const stu = store.students.find(s => s.id === (r && r.student_id)) || {}; const cls = store.classes.find(c => c.id === (r && r.class_id)) || {}; const html = ` Report

STUDENT REPORT CARD

Name: ${stu.first_name || ''} ${stu.last_name || ''}

Admission: ${r.admission_no}

Class: ${cls.name || '-'}

Total${r.total_marks}
Mean${r.mean_score}
Grade${r.grade}
Weak${(r.weak_subjects||[]).join(', ')}
Remark${r.remark}

Printed: ${new Date().toLocaleString()}

`; const w = window.open('', '_blank'); w.document.write(html); w.document.close(); w.print(); }; function applyFilters() { const q = $('searchBox').value.trim().toLowerCase(); const cls = $('classFilter').value; const list = store.reports.filter(r => { const s = store.students.find(x => x.id === r.student_id) || {}; const matchesQ = !q || (s.first_name||'').toLowerCase().includes(q) || (s.last_name||'').toLowerCase().includes(q) || (r.admission_no||'').toLowerCase().includes(q); const matchesC = !cls || r.class_id === cls; return matchesQ && matchesC; }); renderTable(list); } window.addEventListener('load', async () => { await loadAll(); fillClassFilter(); renderTable(store.reports); $('searchBox').oninput = applyFilters; $('classFilter').onchange = applyFilters; $('printAll').onclick = () => window.print(); });
